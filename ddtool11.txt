<script>
let rawData = [];
let headers = [];
let chartInstance = null;

// CSV Upload Handler
csvFile.addEventListener('change', function () {
  const file = this.files[0];
  if (!file) return;

  const reader = new FileReader();
  reader.onload = e => parseCSV(e.target.result);
  reader.readAsText(file);
});

function parseCSV(text) {
  const rows = text.trim().split('\n').map(r => r.split(','));
  headers = rows.shift();
  rawData = rows.map(r => Object.fromEntries(headers.map((h, i) => [h, r[i]])));

  populateSelectors();
  renderTable(rawData);
}

function populateSelectors() {
  columnSelect.innerHTML = '';
  filterColumn.innerHTML = '';

  headers.forEach(h => {
    columnSelect.innerHTML += `<option value="${h}">${h}</option>`;
    filterColumn.innerHTML += `<option value="${h}">${h}</option>`;
  });
}

function analyzeColumn() {
  const col = columnSelect.value;
  const zLimit = parseFloat(zThreshold.value);

  const values = rawData.map(r => parseFloat(r[col])).filter(v => !isNaN(v));
  const mean = values.reduce((a,b)=>a+b,0) / values.length;
  const sd = Math.sqrt(values.reduce((a,b)=>a+Math.pow(b-mean,2),0) / values.length);

  rawData.forEach(r => {
    const v = parseFloat(r[col]);
    r._z = isNaN(v) ? null : (v - mean) / sd;
    r._anomaly = Math.abs(r._z) > zLimit;
  });

  stats.innerHTML = `
    <b>Mean:</b> ${mean.toFixed(3)}<br>
    <b>Standard Deviation:</b> ${sd.toFixed(3)}<br>
    <span class="tooltip" title="Z = (value ‚àí mean) / SD"><b>Z-Score Formula</b></span>
  `;

  renderTable(rawData);
  renderChart(values, col);
}

function renderTable(data) {
  dataTable.innerHTML = '';
  const head = `<tr>${headers.map(h=>`<th>${h}</th>`).join('')}<th>Z</th></tr>`;
  dataTable.innerHTML += head;

  data.forEach(r => {
    const anomalyStyle = r._anomaly ? 'style="background:#ffe6e6"' : '';
    dataTable.innerHTML += `
      <tr ${anomalyStyle}>
        ${headers.map(h=>`<td>${r[h]}</td>`).join('')}
        <td>${r._z ? r._z.toFixed(2) : ''}</td>
      </tr>`;
  });
}

function applyFilter() {
  const col = filterColumn.value;
  const val = filterValue.value;
  const filtered = rawData.filter(r => r[col] == val);
  renderTable(filtered);
}

function renderChart(values, label) {
  if (chartInstance) chartInstance.destroy();
  chartInstance = new Chart(chart, {
    type: 'line',
    data: {
      labels: values.map((_,i)=>i+1),
      datasets: [{
        label: label,
        data: values,
        fill: false
      }]
    }
  });
}
